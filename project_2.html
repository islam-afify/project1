def print_mini_max_values(my_parameter):

    index_lowest = statmov[my_parameter].idxmin()


    index_highest = statmov[my_parameter].idxmax()

    low_value = pd.DataFrame(statmov.loc[index_lowest])
    high_value = pd.DataFrame(statmov.loc[index_highest])

#show the movie with lowest and highest values 
 
    print("The movie with lowest"+ " " + my_parameter + "  " + "  " , statmov['original_title'][index_lowest])

    print("The movie with highest" +  " " + my_parameter + "  " + "  " , statmov['original_title'][index_highest])

    return pd.concat([low_value,high_value],axis = 1)
